# Check Permutation: Given two strings, write a method to decide if one is a permutation of the other. p.90
```python
def check_permutation(s1, s2):
    if len(s1) != len(s2):
        return False
    sd1 = {}
    for i in s1:
        if sd1.get(i):
            sd1[i] = sd1[i] + 1
        else:
            sd1[i] = 1
    for j in s2:
        if sd1.get(j):
            if sd1.get(j) > 0:
                sd1[i] = sd1[i] - 1
            else:
                return False
        else:
            return False
    return True
        
```
